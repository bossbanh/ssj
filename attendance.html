Google Sheet Webhook

วาง URL ของ Google Apps Script Web App (ถ้าปล่อยว่าง ระบบจะเก็บเฉพาะในเครื่อง)
📋 Apps Script Code 
function doPost(e) {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  let sheet = ss.getSheetByName('Attendance');
  if (!sheet) {
    sheet = ss.insertSheet('Attendance');
    sheet.appendRow(['Timestamp','Type','Name','EmployeeID','Department',
      'Note','Latitude','Longitude','Accuracy','MapsLink','Photo','SubmittedBy','Device']);
  }
  const data = JSON.parse(e.postData.contents);
  let photoUrl = '';
  if (data.photo) {
    try {
      const folderName = 'Attendance Photos';
      const folders = DriveApp.getFoldersByName(folderName);
      const folder = folders.hasNext() ? folders.next() : DriveApp.createFolder(folderName);
      const base64 = data.photo.split(',')[1];
      const blob = Utilities.newBlob(Utilities.base64Decode(base64), 'image/jpeg',
        `${data.type}_${data.employeeId||'x'}_${new Date().getTime()}.jpg`);
      const file = folder.createFile(blob);
      file.setSharing(DriveApp.Access.ANYONE_WITH_LINK, DriveApp.Permission.VIEW);
      photoUrl = file.getUrl();
    } catch(err) { photoUrl = 'ERROR: ' + err.message; }
  }
  const mapsLink = data.lat && data.lng ? `https://www.google.com/maps?q=${data.lat},${data.lng}` : '';
  sheet.appendRow([new Date(data.timestamp), data.type, data.name||'',
    data.employeeId||'', data.department||'', data.note||'',
    data.lat||'', data.lng||'', data.accuracy||'', mapsLink, photoUrl,
    data.submittedBy||'', data.device||'']);
  return ContentService.createTextOutput(JSON.stringify({ok:true, photoUrl}))
    .setMimeType(ContentService.MimeType.JSON);
}
function doGet(e) {
  return ContentService.createTextOutput(JSON.stringify({ok:true, message:'Attendance API'}))
    .setMimeType(ContentService.MimeType.JSON);
}
